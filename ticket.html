/* استبدل السكريبت في صفحة ticket.html بهذا */ const session = JSON.parse(localStorage.getItem('wasata_session')); if(!session) location.href='login.html'; // نظام التذكرة الثابتة function initTicket() { let ticketID = localStorage.getItem('wasata_ticket_id'); const uName = session.email.split('@')[0]; // إذا لم يكن لديه تذكرة، أنشئ أول واحدة برقم 16251 if (!ticketID) { ticketID = "#16251"; localStorage.setItem('wasata_ticket_id', ticketID); } else { // إذا دخل الصفحة وعنده تذكر مسبقاً، يروح للشات فوراً startChat(true); } document.getElementById('uName').innerText = uName; document.getElementById('ticketID').innerText = ticketID; } function startChat(isReturning = false) { document.getElementById('setupForm').classList.add('hidden'); document.getElementById('chatArea').classList.remove('hidden'); if(!isReturning) { updateStatus("انتظار دخول الوسيط"); setTimeout(() => { receiveMsg("Waseet_Security", "أهلاً بك يا غالي. أنا الوسيط المسؤول عن هذه العملية. يرجى تأكيد المبلغ المطلوب ووسيلة التسليم."); updateStatus("جاري المراجعة"); }, 2000); } } function sendMsg() { const input = document.getElementById('userMsg'); if(!input.value) return; const msg = input.value; const list = document.getElementById('messageList'); const div = document.createElement('div'); div.className = "msg-user message shadow-xl"; div.innerText = msg; list.appendChild(div); // حفظ الرسالة للأدمن const chatLog = JSON.parse(localStorage.getItem('admin_inbox') || '[]'); chatLog.push({ user: session.email, msg: msg, time: new Date().toLocaleTimeString(), ticket: localStorage.getItem('wasata_ticket_id') }); localStorage.setItem('admin_inbox', JSON.stringify(chatLog)); input.value = ""; list.scrollTop = list.scrollHeight; } // تشغيل عند تحميل الصفحة initTicket(); // الاستماع لتعليمات الأدمن مع التأكد أنها موجهة لهذا المستخدم بالذات window.addEventListener('storage', (e) => { if(e.key === 'admin_instruction') { const data = JSON.parse(e.newValue); // فقط استقبل الرسالة إذا كانت موجهة لإيميلي if(data.targetUser === session.email) { if(data.msg && data.msg !== "طلب دفع مالي") receiveMsg("Waseet_Security", data.msg); if(data.action === 'showPay') document.getElementById('paymentNotice').style.display = 'block'; if(data.status) updateStatus(data.status); } } });
